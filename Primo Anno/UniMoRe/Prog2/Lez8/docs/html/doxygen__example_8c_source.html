92double power(double base, int exponent) {
93 if (exponent == 0) {
94 return 1.0;
95 }
96
97 if (exponent < 0) {
98 return 1.0 / power(base, -exponent);
99 }
100
101 double result = 1.0;
102 double current_power = base;
103
104 while (exponent > 0) {
105 if (exponent % 2 == 1) {
106 result *= current_power;
107 }
108 current_power *= current_power;
109 exponent /= 2;
110 }
111
112 return result;
113}

169int analyze_string(const char *str, StringInfo *info) {
170 if (str == NULL || info == NULL) {
171 return -1;
172 }
173
174 // Inizializza la struttura
175 memset(info, 0, sizeof(StringInfo));
176
177 info->length = strlen(str);
178 if (info->length == 0) {
179 return 0;
180 }
181
182 info->first_char = str[0];
183 info->last_char = str[info->length - 1];
184
185 int in_word = 0;
186
187 for (size_t i = 0; i < info->length; i++) {
188 char c = str[i];
189
190 // Conta parole
191 if (c != ' ' && c != '\t' && c != '\n') {
192 if (!in_word) {
193 info->word_count++;
194 in_word = 1;
195 }
196 } else {
197 in_word = 0;
198 }
199
200 // Conta vocali e consonanti
201 if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) {
202 char lower_c = (c >= 'A' && c <= 'Z') ? c + ('a' - 'A') : c;
203 if (lower_c == 'a' || lower_c == 'e' || lower_c == 'i' ||
204 lower_c == 'o' || lower_c == 'u') {
205 info->vowel_count++;
206 } else {
207 info->consonant_count++;
208 }
209 }
210 }
211
212 return 0;
213}

262void bubble_sort(int arr[], size_t size, SortOrder order) {
263 if (arr == NULL || size <= 1) {
264 return;
265 }
266
267 for (size_t i = 0; i < size - 1; i++) {
268 int swapped = 0;
269
270 for (size_t j = 0; j < size - i - 1; j++) {
271 int should_swap = 0;
272
273 if (order == ASCENDING) {
274 should_swap = (arr[j] > arr[j + 1]);
275 } else {
276 should_swap = (arr[j] < arr[j + 1]);
277 }
278
279 if (should_swap) {
280 // Scambia gli elementi
281 int temp = arr[j];
282 arr[j] = arr[j + 1];
283 arr[j + 1] = temp;
284 swapped = 1;
285 }
286 }
287
288 // Se non ci sono stati scambi, l'array √® gi√† ordinato
289 if (!swapped) {
290 break;
291 }
292 }
293}

320int binary_search(const int arr[], size_t size, int target) {
321 if (arr == NULL || size == 0) {
322 return -1;
323 }
324
325 int left = 0;
326 int right = (int)size - 1;
327
328 while (left <= right) {
329 int mid = left + (right - left) / 2;
330
331 if (arr[mid] == target) {
332 return mid;
333 } else if (arr[mid] < target) {
334 left = mid + 1;
335 } else {
336 right = mid - 1;
337 }
338 }
339
340 return -1; // Elemento non trovato
341}
